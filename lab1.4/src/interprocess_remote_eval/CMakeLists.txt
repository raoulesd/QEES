function(custom_executable target)
  add_executable(${target}${target_suffix}
    src/${target}.cpp)
  ament_target_dependencies(${target}${target_suffix}
    "rclcpp"
    "std_msgs"
	"rt")
  install(TARGETS ${target}${target_suffix}
    DESTINATION lib/${PROJECT_NAME})
endfunction()

endmacro()
